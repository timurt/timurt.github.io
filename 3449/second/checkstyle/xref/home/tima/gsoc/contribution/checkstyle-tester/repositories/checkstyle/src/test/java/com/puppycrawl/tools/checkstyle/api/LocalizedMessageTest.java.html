65      @Test
66      public void testBundleReloadUrlNull() throws IOException {
67          final LocalizedMessage.Utf8Control control = new LocalizedMessage.Utf8Control();
68          control.newBundle("com.puppycrawl.tools.checkstyle.checks.coding.messages",
69                  Locale.ENGLISH, "java.class",
70                  Thread.currentThread().getContextClassLoader(), true);
71      }
72  
73      @Test
74      public void testBundleReloadUrlNotNull() throws IOException {
75  
76          final ClassLoader classloader = mock(ClassLoader.class);
77          final URLConnection mockConnection = Mockito.mock(URLConnection.class);
78          when(mockConnection.getInputStream()).thenReturn(
79                  new ByteArrayInputStream(EMPTY_BYTE_ARRAY));
80  
81          final URL url = getMockUrl(mockConnection);
82          final String resource =
83              "com/puppycrawl/tools/checkstyle/checks/coding/messages_en.properties";
84          when(classloader.getResource(resource)).thenReturn(url);
85  
86          final LocalizedMessage.Utf8Control control = new LocalizedMessage.Utf8Control();
87          control.newBundle("com.puppycrawl.tools.checkstyle.checks.coding.messages",
88                  Locale.ENGLISH, "java.class",
89                  classloader, true);
90      }
91  
92      @Test
93      public void testBundleReloadUrlNotNullStreamNull() throws IOException {
94  
95          final ClassLoader classloader = mock(ClassLoader.class);
96          final String resource =
97              "com/puppycrawl/tools/checkstyle/checks/coding/messages_en.properties";
98  
99          final URL url = getMockUrl(null);
100         when(classloader.getResource(resource)).thenReturn(url);
101 
102         final LocalizedMessage.Utf8Control control = new LocalizedMessage.Utf8Control();
103         control.newBundle("com.puppycrawl.tools.checkstyle.checks.coding.messages",
104                 Locale.ENGLISH, "java.class",
105                 classloader, true);
106     }
107 
108     private static URL getMockUrl(final URLConnection connection) throws IOException {
109         final URLStreamHandler handler = new URLStreamHandler() {
110             @Override
111             protected URLConnection openConnection(final URL url) {
112                 return connection;
113             }
114         };
115         return new URL("http://foo.bar", "foo.bar", 80, "", handler);
116     }
