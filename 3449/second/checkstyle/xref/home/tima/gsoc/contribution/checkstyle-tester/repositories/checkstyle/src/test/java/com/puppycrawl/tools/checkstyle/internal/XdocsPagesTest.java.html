162     private static void buildAndValidateXml(String fileName, String unserializedSource)
163             throws IOException, ParserConfigurationException, CheckstyleException {
164         // not all examples come with the full xml structure
165         String code = unserializedSource
166             // don't corrupt our own cachefile
167             .replace("target/cachefile", "target/cachefile-test");
168 
169         if (!hasFileSetClass(code)) {
170             code = "<module name=\"TreeWalker\">\n" + code + "\n</module>";
171         }
172         if (!code.contains("name=\"Checker\"")) {
173             code = "<module name=\"Checker\">\n" + code + "\n</module>";
174         }
175         if (!code.startsWith("<?xml")) {
176             final String dtdPath = new File(
177                     "src/main/resources/com/puppycrawl/tools/checkstyle/configuration_1_3.dtd")
178                     .getCanonicalPath();
179 
180             code = "<?xml version=\"1.0\"?>\n<!DOCTYPE module PUBLIC "
181                     + "\"-//Puppy Crawl//DTD Check Configuration 1.3//EN\" \"" + dtdPath + "\">\n"
182                     + code;
183         }
184 
185         // validate only
186         XmlUtil.getRawXml(fileName, code, unserializedSource);
187 
188         // can't test ant structure, or old and outdated checks
189         if (!fileName.startsWith("anttask") && !fileName.startsWith("releasenotes")) {
190             validateCheckstyleXml(fileName, code, unserializedSource);
191         }
192     }

194     private static boolean hasFileSetClass(String xml) {
195         boolean found = false;
196 
197         for (String find : XML_FILESET_LIST) {
198             if (xml.contains(find)) {
199                 found = true;
200                 break;
201             }
202         }
203 
204         return found;
205     }

207     private static void validateCheckstyleXml(String fileName, String code,
208             String unserializedSource) throws IOException, CheckstyleException {
209         // can't process non-existent examples, or out of context snippets
210         if (!code.contains("com.mycompany") && !code.contains("checkstyle-packages")
211                 && !code.contains("MethodLimit") && !code.contains("<suppress ")
212                 && !code.contains("<import-control ")
213                 && !unserializedSource.startsWith("<property ")
214                 && !unserializedSource.startsWith("<taskdef ")) {
215             // validate checkstyle structure and contents
216             try {
217                 final Properties properties = new Properties();
218 
219                 properties.setProperty("checkstyle.header.file",
220                         new File("config/java.header").getCanonicalPath());
221 
222                 final PropertiesExpander expander = new PropertiesExpander(properties);
223                 final Configuration config = ConfigurationLoader.loadConfiguration(new InputSource(
224                         new StringReader(code)), expander, false);
225                 final Checker checker = new Checker();
226 
227                 try {
228                     final ClassLoader moduleClassLoader = Checker.class.getClassLoader();
229                     checker.setModuleClassLoader(moduleClassLoader);
230                     checker.configure(config);
231                 }
232                 finally {
233                     checker.destroy();
234                 }
235             }
236             catch (CheckstyleException ex) {
237                 throw new CheckstyleException(fileName + " has invalid Checkstyle xml ("
238                         + ex.getMessage() + "): " + unserializedSource, ex);
239             }
240         }
241     }

246 
247         for (Path path : XdocUtil.getXdocsConfigFilePaths(XdocUtil.getXdocsFilePaths())) {
248             final String fileName = path.getFileName().toString();
249 
250             if ("config_reporting.xml".equals(fileName)) {

252             }
253 
254             final String input = new String(Files.readAllBytes(path), UTF_8);
255             final Document document = XmlUtil.getRawXml(fileName, input, input);
256             final NodeList sources = document.getElementsByTagName("section");
257             String lastSectioName = null;
258 
259             for (int position = 0; position < sources.getLength(); position++) {
260                 final Node section = sources.item(position);
261                 final String sectionName = section.getAttributes().getNamedItem("name")
262                         .getNodeValue();
263 
264                 if ("Content".equals(sectionName) || "Overview".equals(sectionName)) {
265                     Assert.assertNull(fileName + " section '" + sectionName + "' should be first",
266                             lastSectioName);
267                     continue;
268                 }
269 
270                 Assert.assertTrue(fileName + " section '" + sectionName
271                         + "' shouldn't end with 'Check'", !sectionName.endsWith("Check"));
272                 if (lastSectioName != null) {
273                     Assert.assertTrue(
274                             fileName + " section '" + sectionName
275                                     + "' is out of order compared to '" + lastSectioName + "'",
276                             sectionName.toLowerCase(Locale.ENGLISH).compareTo(
277                                     lastSectioName.toLowerCase(Locale.ENGLISH)) >= 0);
278                 }
279 
280                 validateCheckSection(moduleFactory, fileName, sectionName, section);
281 
282                 lastSectioName = sectionName;
283             }
284         }

311     private static void validateCheckSection(ModuleFactory moduleFactory, String fileName,
312             String sectionName, Node section) throws Exception {
313         final Object instance;
314 
315         try {
316             instance = moduleFactory.createModule(sectionName);
317         }
318         catch (CheckstyleException ex) {
319             throw new CheckstyleException(fileName + " couldn't find class: " + sectionName, ex);
320         }
321 
322         int subSectionPos = 0;
323         for (Node subSection : XmlUtil.getChildrenElements(section)) {
324             final String subSectionName = subSection.getAttributes().getNamedItem("name")
325                     .getNodeValue();
326 
327             // can be in different orders, and completely optional
328             if ("Notes".equals(subSectionName)
329                     || "Rule Description".equals(subSectionName)) {
330                 continue;
331             }
332 
333             // optional sections that can be skipped if they have nothing to report
334             if (subSectionPos == 1 && !"Properties".equals(subSectionName)) {
335                 validatePropertySection(fileName, sectionName, null, instance);
336                 subSectionPos++;
337             }
338             if (subSectionPos == 4 && !"Error Messages".equals(subSectionName)) {
339                 validateErrorSection(fileName, sectionName, null, instance);
340                 subSectionPos++;
341             }
342 
343             Assert.assertEquals(fileName + " section '" + sectionName
344                     + "' should be in order", getSubSectionName(subSectionPos),
345                     subSectionName);
346 
347             switch (subSectionPos) {
348                 case 0:
349                     break;
350                 case 1:
351                     validatePropertySection(fileName, sectionName, subSection, instance);
352                     break;
353                 case 2:
354                     break;
355                 case 3:
356                     validateUsageExample(fileName, sectionName, subSection);
357                     break;
358                 case 4:
359                     validateErrorSection(fileName, sectionName, subSection, instance);
360                     break;
361                 case 5:
362                     validatePackageSection(fileName, sectionName, subSection, instance);
363                     break;
364                 case 6:
365                     validateParentSection(fileName, sectionName, subSection);
366                     break;
367                 default:
368                     break;
369             }
370 
371             subSectionPos++;
372         }
373     }

375     private static Object getSubSectionName(int subSectionPos) {
376         final String result;
377 
378         switch (subSectionPos) {
379             case 0:
380                 result = "Description";
381                 break;
382             case 1:
383                 result = "Properties";
384                 break;
385             case 2:
386                 result = "Examples";
387                 break;
388             case 3:
389                 result = "Example of Usage";
390                 break;
391             case 4:
392                 result = "Error Messages";
393                 break;
394             case 5:
395                 result = "Package";
396                 break;
397             case 6:
398                 result = "Parent Module";
399                 break;
400             default:
401                 result = null;
402                 break;
403         }
404 
405         return result;
406     }

408     private static void validatePropertySection(String fileName, String sectionName,
409             Node subSection, Object instance) throws Exception {
410         final Set<String> properties = getProperties(instance.getClass());
411         final Class<?> clss = instance.getClass();
412 
413         fixCapturedProperties(sectionName, instance, clss, properties);
414 
415         if (subSection != null) {
416             Assert.assertTrue(fileName + " section '" + sectionName
417                     + "' should have no properties to show", !properties.isEmpty());
418 
419             validatePropertySectionProperties(fileName, sectionName, subSection, instance,
420                     properties);
421         }
422 
423         Assert.assertTrue(fileName + " section '" + sectionName + "' should show properties: "
424                 + properties, properties.isEmpty());
425     }

427     private static void fixCapturedProperties(String sectionName, Object instance, Class<?> clss,
428             Set<String> properties) {
429         // remove global properties that don't need documentation
430         if (hasParentModule(sectionName)) {
431             if (AbstractJavadocCheck.class.isAssignableFrom(clss)) {
432                 properties.removeAll(JAVADOC_CHECK_PROPERTIES);
433             }
434             else if (AbstractCheck.class.isAssignableFrom(clss)) {
435                 properties.removeAll(CHECK_PROPERTIES);
436             }
437         }
438         if (AbstractFileSetCheck.class.isAssignableFrom(clss)) {
439             properties.removeAll(FILESET_PROPERTIES);
440 
441             // override
442             properties.add("fileExtensions");
443         }
444 
445         // remove undocumented properties
446         new HashSet<>(properties).stream()
447             .filter(p -> UNDOCUMENTED_PROPERTIES.contains(clss.getSimpleName() + "." + p))
448             .forEach(properties::remove);
449 
450         if (AbstractCheck.class.isAssignableFrom(clss)) {
451             final AbstractCheck check = (AbstractCheck) instance;
452 
453             final int[] acceptableTokens = check.getAcceptableTokens();
454             Arrays.sort(acceptableTokens);
455             final int[] defaultTokens = check.getDefaultTokens();
456             Arrays.sort(defaultTokens);
457             final int[] requiredTokens = check.getRequiredTokens();
458             Arrays.sort(requiredTokens);
459 
460             if (!Arrays.equals(acceptableTokens, defaultTokens)
461                     || !Arrays.equals(acceptableTokens, requiredTokens)) {
462                 properties.add("tokens");
463             }
464         }
465 
466         if (AbstractJavadocCheck.class.isAssignableFrom(clss)) {
467             final AbstractJavadocCheck check = (AbstractJavadocCheck) instance;
468 
469             final int[] acceptableJavadocTokens = check.getAcceptableJavadocTokens();
470             Arrays.sort(acceptableJavadocTokens);
471             final int[] defaultJavadocTokens = check.getDefaultJavadocTokens();
472             Arrays.sort(defaultJavadocTokens);
473             final int[] requiredJavadocTokens = check.getRequiredJavadocTokens();
474             Arrays.sort(requiredJavadocTokens);
475 
476             if (!Arrays.equals(acceptableJavadocTokens, defaultJavadocTokens)
477                     || !Arrays.equals(acceptableJavadocTokens, requiredJavadocTokens)) {
478                 properties.add("javadocTokens");
479             }
480         }
481     }

483     private static void validatePropertySectionProperties(String fileName, String sectionName,
484             Node subSection, Object instance, Set<String> properties) throws Exception {
485         boolean skip = true;
486         boolean didJavadocTokens = false;
487         boolean didTokens = false;
488 
489         for (Node row : XmlUtil.getChildrenElements(XmlUtil.getFirstChildElement(subSection))) {
490             if (skip) {
491                 skip = false;
492                 continue;
493             }
494             Assert.assertFalse(fileName + " section '" + sectionName
495                     + "' should have token properties last", didTokens);
496 
497             final List<Node> columns = new ArrayList<>(XmlUtil.getChildrenElements(row));
498 
499             final String propertyName = columns.get(0).getTextContent();
500             Assert.assertTrue(fileName + " section '" + sectionName
501                     + "' should not contain the property: " + propertyName,
502                     properties.remove(propertyName));
503 
504             if ("tokens".equals(propertyName)) {
505                 final AbstractCheck check = (AbstractCheck) instance;
506                 validatePropertySectionPropertyTokens(fileName, sectionName, check, columns);
507                 didTokens = true;
508             }
509             else if ("javadocTokens".equals(propertyName)) {
510                 final AbstractJavadocCheck check = (AbstractJavadocCheck) instance;
511                 validatePropertySectionPropertyJavadocTokens(fileName, sectionName, check, columns);
512                 didJavadocTokens = true;
513             }
514             else {
515                 Assert.assertFalse(fileName + " section '" + sectionName
516                         + "' should have javadoc token properties next to last, before tokens",
517                         didJavadocTokens);
518                 Assert.assertFalse(fileName + " section '" + sectionName
519                         + "' should have a description for " + propertyName, columns.get(1)
520                         .getTextContent().trim().isEmpty());
521 
522                 final String actualTypeName = columns.get(2).getTextContent().replace("\n", "")
523                         .replace("\r", "").replaceAll(" +", " ").trim();
524                 final String actualValue = columns.get(3).getTextContent().replace("\n", "")
525                         .replace("\r", "").replaceAll(" +", " ").trim();
526 
527                 Assert.assertFalse(fileName + " section '" + sectionName
528                         + "' should have a type for " + propertyName, actualTypeName.isEmpty());
529 
530                 final PropertyDescriptor descriptor = PropertyUtils.getPropertyDescriptor(instance,
531                         propertyName);
532                 final Class<?> clss = descriptor.getPropertyType();
533                 final String expectedTypeName =
534                         getModulePropertyExpectedTypeName(clss, instance, propertyName);
535                 final String expectedValue = getModulePropertyExpectedValue(clss, instance,
536                         propertyName);
537 
538                 if (expectedTypeName != null) {
539                     Assert.assertEquals(fileName + " section '" + sectionName
540                             + "' should have the type for " + propertyName, expectedTypeName,
541                             actualTypeName);
542                     if (expectedValue != null) {
543                         Assert.assertEquals(fileName + " section '" + sectionName
544                                 + "' should have the value for " + propertyName, expectedValue,
545                                 actualValue);
546                     }
547                 }
548             }
549         }
550     }

589     /** @noinspection IfStatementWithTooManyBranches */
590     private static String getModulePropertyExpectedTypeName(Class<?> clss, Object instance,
591             String propertyName) {
592         final String instanceName = instance.getClass().getSimpleName();
593         String result = null;
594 
595         if (clss == boolean.class) {
596             result = "Boolean";
597         }
598         else if (clss == int.class) {
599             result = "Integer";
600         }
601         else if (clss == int[].class) {
602             result = "Integer Set";
603         }
604         else if (clss == double[].class) {
605             result = "Number Set";
606         }
607         else if (clss == String[].class) {
608             if (propertyName.endsWith("Tokens") || propertyName.endsWith("Token")
609                     || "AtclauseOrderCheck".equals(instanceName) && "target".equals(propertyName)
610                     || "MultipleStringLiteralsCheck".equals(instanceName)
611                             && "ignoreOccurrenceContext".equals(propertyName)) {
612                 result = "subset of tokens TokenTypes";
613             }
614             else {
615                 result = "String Set";
616             }
617         }
618         else if (clss == URI.class) {
619             result = "URI";
620         }
621         else if (clss == Pattern.class) {
622             result = "Regular Expression";
623         }
624         else if (clss == SeverityLevel.class) {
625             result = "Severity";
626         }
627         else if (clss == Scope.class) {
628             result = "Scope";
629         }
630         else if (clss == AccessModifier[].class) {
631             result = "Access Modifier Set";
632         }
633         else if (clss != String.class) {
634             Assert.fail("Unknown property type: " + clss.getSimpleName());
635         }
636 
637         if ("SuppressWarningsHolder".equals(instanceName)) {
638             result = result + " in a format of comma separated attribute=value entries. The "
639                     + "attribute is the fully qualified name of the Check and value is its alias.";
640         }
641 
642         return result;
643     }

645     private static String getModulePropertyExpectedValue(Class<?> clss, Object instance,
646             String propertyName) throws Exception {
647         final Field field = getField(instance.getClass(), propertyName);
648         String result = null;
649 
650         if (field != null) {
651             final Object value = field.get(instance);
652 
653             if (clss == boolean.class) {
654                 result = value.toString();
655             }
656             else if (clss == int.class) {
657                 if (value.equals(Integer.MAX_VALUE)) {
658                     result = "java.lang.Integer.MAX_VALUE";
659                 }
660                 else {
661                     result = value.toString();
662                 }
663             }
664             else if (clss == int[].class) {
665                 result = Arrays.toString((int[]) value).replace("[", "").replace("]", "");
666                 if (result.isEmpty()) {
667                     result = "{}";
668                 }
669             }
670             else if (clss == double[].class) {
671                 result = Arrays.toString((double[]) value).replace("[", "").replace("]", "")
672                         .replace(".0", "");
673                 if (result.isEmpty()) {
674                     result = "{}";
675                 }
676             }
677             else if (clss == URI.class) {
678                 if (value != null) {
679                     result = '"' + value.toString() + '"';
680                 }
681             }
682             else if (clss == Pattern.class) {
683                 if (value != null) {
684                     result = '"' + value.toString().replace("\n", "\\n").replace("\t", "\\t")
685                             .replace("\r", "\\r").replace("\f", "\\f") + '"';
686                 }
687 
688                 if ("\"$^\"".equals(result)) {
689                     result += " (empty)";
690                 }
691             }
692             else if (value != null && (clss == SeverityLevel.class || clss == Scope.class)) {
693                 result = value.toString().toLowerCase(Locale.ENGLISH);
694             }
695             else if (value != null && clss == AccessModifier[].class) {
696                 result = Arrays.toString((Object[]) value).replace("[", "").replace("]", "");
697             }
698 
699             if (clss != String.class && clss != String[].class && result == null) {
700                 result = "null";
701             }
702         }
703 
704         return result;
705     }

707     private static Field getField(Class<?> clss, String propertyName) {
708         Field result = null;
709 
710         if (clss != null) {
711             try {
712                 result = clss.getDeclaredField(propertyName);
713                 result.setAccessible(true);
714             }
715             catch (NoSuchFieldException ex) {
716                 result = getField(clss.getSuperclass(), propertyName);
717             }
718         }
719 
720         return result;
721     }

723     private static void validateErrorSection(String fileName, String sectionName, Node subSection,
724             Object instance) throws Exception {

726         final Set<Field> fields = CheckUtil.getCheckMessages(clss);
727         final Set<String> list = new TreeSet<>();
728

750         if (subSection == null) {

752                     + "' should have the expected error keys", "", expectedText.toString());
753         }
754         else {
755             Assert.assertEquals(fileName + " section '" + sectionName
756                     + "' should have the expected error keys", expectedText.toString().trim(),
757                     subSection.getTextContent().replaceAll("\n\\s+", "\n").trim());
758 
759             for (Node node : XmlUtil.findChildElementsByTag(subSection, "a")) {
760                 final String url = node.getAttributes().getNamedItem("href").getTextContent();
761                 final String linkText = node.getTextContent().trim();
762                 final String expectedUrl;

769                             + "path%3Asrc%2Fmain%2Fresources%2F"
770                             + clss.getPackage().getName().replace(".", "%2F")
771                             + "+filename%3Amessages*.properties+repo%3Acheckstyle%2Fcheckstyle+%22"
772                             + linkText + "%22";
773                 }
774 
775                 Assert.assertEquals(fileName + " section '" + sectionName
776                         + "' should have matching url for '" + linkText + "'", expectedUrl, url);
777             }
778         }

892     public void testAllStyleRules() throws Exception {
893         for (Path path : XdocUtil.getXdocsStyleFilePaths(XdocUtil.getXdocsFilePaths())) {
894             final String fileName = path.getFileName().toString();
895             final String input = new String(Files.readAllBytes(path), UTF_8);
896             final Document document = XmlUtil.getRawXml(fileName, input, input);
897             final NodeList sources = document.getElementsByTagName("tr");
898             Set<String> styleChecks = null;
899 
900             if (path.toFile().getName().contains("google")) {
901                 styleChecks = new HashSet<>(GOOGLE_MODULES);
902             }
903             else if (path.toFile().getName().contains("sun")) {
904                 styleChecks = new HashSet<>();
905             }
906 
907             String lastRuleName = null;
908 
909             for (int position = 0; position < sources.getLength(); position++) {
910                 final Node row = sources.item(position);
911                 final List<Node> columns = new ArrayList<>(
912                         XmlUtil.findChildElementsByTag(row, "td"));
913 
914                 if (columns.isEmpty()) {
915                     continue;
916                 }
917 
918                 final String ruleName = columns.get(1).getTextContent().trim();
919 
920                 if (lastRuleName != null) {
921                     Assert.assertTrue(
922                             fileName + " rule '" + ruleName + "' is out of order compared to '"
923                                     + lastRuleName + "'",
924                             ruleName.toLowerCase(Locale.ENGLISH).compareTo(
925                                     lastRuleName.toLowerCase(Locale.ENGLISH)) >= 0);
926                 }
927 
928                 if (!"--".equals(ruleName)) {
929                     validateStyleAnchors(XmlUtil.findChildElementsByTag(columns.get(0), "a"),
930                             fileName, ruleName);
931                 }
932 
933                 validateStyleModules(XmlUtil.findChildElementsByTag(columns.get(2), "a"),
934                         XmlUtil.findChildElementsByTag(columns.get(3), "a"), styleChecks, fileName,
935                         ruleName);
936 
937                 lastRuleName = ruleName;
938             }
939 
940             // these modules aren't documented, but are added to the config
941             styleChecks.remove("TreeWalker");
942             styleChecks.remove("Checker");
943 
944             Assert.assertTrue(fileName + " requires the following check(s) to appear: "
945                     + styleChecks, styleChecks.isEmpty());
946         }
947     }

952 
953         final int space = ruleName.indexOf(' ');

982     private static void validateStyleModules(Set<Node> checks, Set<Node> configs,
983             Set<String> styleChecks, String fileName, String ruleName) {
984         final Iterator<Node> itrChecks = checks.iterator();
985         final Iterator<Node> itrConfigs = configs.iterator();
986 
987         while (itrChecks.hasNext()) {
988             final Node module = itrChecks.next();
989             final String moduleName = module.getTextContent().trim();
990 
991             if (!module.getAttributes().getNamedItem("href").getTextContent()
992                     .startsWith("config_")) {
993                 continue;
994             }
995 
996             Assert.assertTrue(fileName + " rule '" + ruleName + "' module '" + moduleName
997                     + "' shouldn't end with 'Check'", !moduleName.endsWith("Check"));
998 
999             styleChecks.remove(moduleName);
1000 
1001             for (String configName : new String[] {"config", "test"}) {
1002                 Node config = null;
1003 
1004                 try {
1005                     config = itrConfigs.next();
1006                 }
1007                 catch (NoSuchElementException ignore) {
1008                     Assert.fail(fileName + " rule '" + ruleName + "' module '" + moduleName
1009                             + "' is missing the config link: " + configName);
1010                 }
1011 
1012                 Assert.assertEquals(fileName + " rule '" + ruleName + "' module '" + moduleName
1013                         + "' has mismatched config/test links", configName, config.getTextContent()
1014                         .trim());
1015 
1016                 final String configUrl = config.getAttributes().getNamedItem("href")
1017                         .getTextContent();
1018 
1019                 if ("config".equals(configName)) {
1020                     final String expectedUrl = "https://github.com/search?q="
1021                             + "path%3Asrc%2Fmain%2Fresources+filename%3Agoogle_checks.xml+"
1022                             + "repo%3Acheckstyle%2Fcheckstyle+" + moduleName;
1023 
1024                     Assert.assertEquals(fileName + " rule '" + ruleName + "' module '" + moduleName
1025                             + "' should have matching " + configName + " url", expectedUrl,
1026                             configUrl);
1027                 }
1028                 else if ("test".equals(configName)) {
1029                     Assert.assertTrue(fileName + " rule '" + ruleName + "' module '" + moduleName
1030                             + "' should have matching " + configName + " url",
1031                             configUrl.startsWith("https://github.com/checkstyle/checkstyle/"
1032                                     + "blob/master/src/it/java/com/google/checkstyle/test/"));
1033                     Assert.assertTrue(fileName + " rule '" + ruleName + "' module '" + moduleName
1034                             + "' should have matching " + configName + " url",
1035                             configUrl.endsWith("/" + moduleName + "Test.java"));
1036 
1037                     Assert.assertTrue(fileName + " rule '" + ruleName + "' module '" + moduleName
1038                             + "' should have a test that exists", new File(configUrl.substring(53)
1039                             .replace('/', File.separatorChar)).exists());
1040                 }
1041             }
1042         }
1043 
1044         Assert.assertFalse(fileName + " rule '" + ruleName + "' has too many configs",
1045                 itrConfigs.hasNext());
1046     }
