197     private static DefaultConfiguration createFilterConfig(Class<?> classObj) {
198         return new DefaultConfiguration(classObj.getName());

201     private void verifySuppressed(Configuration filterConfig,
202             String... suppressed)
203             throws Exception {
204         verify(createChecker(filterConfig),
205                getPath("InputSuppressWithNearbyCommentFilter.java"),
206                removeSuppressed(ALL_MESSAGES, suppressed));
207     }

209     @Override
210     public Checker createChecker(Configuration checkConfig)
211             throws CheckstyleException {
212         final DefaultConfiguration checkerConfig =
213             new DefaultConfiguration("configuration");
214         final DefaultConfiguration checksConfig = createCheckConfig(TreeWalker.class);
215         checksConfig.addChild(createCheckConfig(FileContentsHolder.class));
216         final DefaultConfiguration memberNameCheckConfig = createCheckConfig(MemberNameCheck.class);
217         memberNameCheckConfig.addAttribute("id", "ignore");
218         checksConfig.addChild(memberNameCheckConfig);
219         final DefaultConfiguration constantNameCheckConfig =
220             createCheckConfig(ConstantNameCheck.class);
221         constantNameCheckConfig.addAttribute("id", null);
222         checksConfig.addChild(constantNameCheckConfig);
223         checksConfig.addChild(createCheckConfig(IllegalCatchCheck.class));
224         checkerConfig.addChild(checksConfig);
225         if (checkConfig != null) {
226             checkerConfig.addChild(checkConfig);
227         }
228         final Checker checker = new Checker();
229         final Locale locale = Locale.ROOT;
230         checker.setLocaleCountry(locale.getCountry());
231         checker.setLocaleLanguage(locale.getLanguage());
232         checker.setModuleClassLoader(Thread.currentThread().getContextClassLoader());
233         checker.configure(checkerConfig);
234         checker.addListener(new BriefUtLogger(stream));
235         return checker;
236     }

238     private static String[] removeSuppressed(String[] from, String... remove) {
239         final Collection<String> coll = Arrays.stream(from).collect(Collectors.toList());
240         coll.removeAll(Arrays.asList(remove));
241         return coll.toArray(new String[coll.size()]);
242     }

244     @Test
245     public void testInvalidInfluenceFormat() throws Exception {
246         final DefaultConfiguration filterConfig =
247             createFilterConfig(SuppressWithNearbyCommentFilter.class);
248         filterConfig.addAttribute("influenceFormat", "a");
249 
250         try {
251             final String[] suppressed = CommonUtils.EMPTY_STRING_ARRAY;
252             verifySuppressed(filterConfig, suppressed);
253         }
254         catch (CheckstyleException ex) {
255             final IllegalArgumentException cause = (IllegalArgumentException) ex.getCause();
256             assertEquals("unable to parse influence"
257                             + " from 'SUPPRESS CHECKSTYLE MemberNameCheck' using a",
258                     cause.getMessage());
259         }
260     }

307         final DefaultConfiguration filterConfig =
308             createFilterConfig(SuppressWithNearbyCommentFilter.class);
309         filterConfig.addAttribute("commentFormat", "@cs-: (\\w+) \\(\\w+\\)");
