32  /**
33   * <p>
34   * Checks the placement of right curly braces.
35   * The policy to verify is specified using the {@link RightCurlyOption} class
36   * and defaults to {@link RightCurlyOption#SAME}.
37   * </p>
38   * <p> By default the check will check the following tokens:
39   *  {@link TokenTypes#LITERAL_TRY LITERAL_TRY},
40   *  {@link TokenTypes#LITERAL_CATCH LITERAL_CATCH},
41   *  {@link TokenTypes#LITERAL_FINALLY LITERAL_FINALLY},
42   *  {@link TokenTypes#LITERAL_IF LITERAL_IF},
43   *  {@link TokenTypes#LITERAL_ELSE LITERAL_ELSE}.
44   * Other acceptable tokens are:
45   *  {@link TokenTypes#CLASS_DEF CLASS_DEF},
46   *  {@link TokenTypes#METHOD_DEF METHOD_DEF},
47   *  {@link TokenTypes#CTOR_DEF CTOR_DEF}.
48   *  {@link TokenTypes#LITERAL_FOR LITERAL_FOR}.
49   *  {@link TokenTypes#LITERAL_WHILE LITERAL_WHILE}.
50   *  {@link TokenTypes#LITERAL_DO LITERAL_DO}.
51   *  {@link TokenTypes#STATIC_INIT STATIC_INIT}.
52   *  {@link TokenTypes#INSTANCE_INIT INSTANCE_INIT}.
53   * </p>
54   * <p>
55   * <b>shouldStartLine</b> - does the check need to check
56   * if right curly starts line. Default value is <b>true</b>
57   * </p>
58   * <p>
59   * An example of how to configure the check is:
60   * </p>
61   * <pre>
62   * &lt;module name="RightCurly"/&gt;
63   * </pre>
64   * <p>
65   * An example of how to configure the check with policy
66   * {@link RightCurlyOption#ALONE} for {@code else} and
67   * {@code {@link TokenTypes#METHOD_DEF METHOD_DEF}}tokens is:
68   * </p>
69   * <pre>
70   * &lt;module name="RightCurly"&gt;
71   *     &lt;property name="tokens" value="LITERAL_ELSE"/&gt;
72   *     &lt;property name="option" value="alone"/&gt;
73   * &lt;/module&gt;
74   * </pre>
75   *
76   * @author Oliver Burn
77   * @author lkuehne
78   * @author o_sukhodolsky
79   * @author maxvetrenko
80   * @author Andrei Selkin
81   * @author <a href="mailto:piotr.listkiewicz@gmail.com">liscju</a>
82   */
83  public class RightCurlyCheck extends AbstractCheck {
84      /**
85       * A key is pointing to the warning message text in "messages.properties"
86       * file.
87       */
88      public static final String MSG_KEY_LINE_BREAK_BEFORE = "line.break.before";
89  
90      /**
91       * A key is pointing to the warning message text in "messages.properties"
92       * file.
93       */
94      public static final String MSG_KEY_LINE_ALONE = "line.alone";
95  
96      /**
97       * A key is pointing to the warning message text in "messages.properties"
98       * file.
99       */
100     public static final String MSG_KEY_LINE_SAME = "line.same";
101 
102     /**
103      * A key is pointing to the warning message text in "messages.properties"
104      * file.
105      */
106     public static final String MSG_KEY_LINE_NEW = "line.new";
107 
108     /** Do we need to check if right curly starts line. */
109     private boolean shouldStartLine = true;
110 
111     /** The policy to enforce. */
112     private RightCurlyOption option = RightCurlyOption.SAME;
113 
114     /**
115      * Sets the option to enforce.
116      * @param optionStr string to decode option from
117      * @throws IllegalArgumentException if unable to decode
118      */
119     public void setOption(String optionStr) {
120         try {
121             option = RightCurlyOption.valueOf(optionStr.trim().toUpperCase(Locale.ENGLISH));
122         }
123         catch (IllegalArgumentException iae) {
124             throw new IllegalArgumentException("unable to parse " + optionStr, iae);
125         }
126     }
127 
128     /**
129      * Does the check need to check if right curly starts line.
130      * @param flag new value of this property.
131      */
132     public void setShouldStartLine(boolean flag) {
133         shouldStartLine = flag;
134     }

140             TokenTypes.LITERAL_CATCH,

171     @Override
172     public void visitToken(DetailAST ast) {
173         final Details details = getDetails(ast);
174         final DetailAST rcurly = details.rcurly;
175 
176         if (rcurly != null) {
177             final String violation = validate(details);
178             if (!violation.isEmpty()) {
179                 log(rcurly, violation, "}", rcurly.getColumnNo() + 1);
180             }
181         }
182     }
183 
184     /**
185      * Does general validation.
186      * @param details for validation.
187      * @return violation message or empty string
188      *     if there was not violation during validation.
189      */
190     private String validate(Details details) {
191         final DetailAST rcurly = details.rcurly;
192         final DetailAST nextToken = details.nextToken;
193         final boolean shouldCheckLastRcurly = details.shouldCheckLastRcurly;
194         String violation = "";
195         if (shouldHaveLineBreakBefore(option, details)) {
196             violation = MSG_KEY_LINE_BREAK_BEFORE;
197         }
198         else if (shouldCheckLastRcurly
199                  && option != RightCurlyOption.ALONE) {
200             if (rcurly.getLineNo() == nextToken.getLineNo()) {
201                 violation = MSG_KEY_LINE_ALONE;
202             }
203         }
204         else if (shouldBeOnSameLine(option, details)) {
205             violation = MSG_KEY_LINE_SAME;
206         }
207         else if (shouldBeAloneOnLine(option, details)) {
208             violation = MSG_KEY_LINE_ALONE;
209         }
210         else if (shouldStartLine) {
211             final String targetSourceLine = getLines()[rcurly.getLineNo() - 1];
212             if (!isOnStartOfLine(details, targetSourceLine)) {
213                 violation = MSG_KEY_LINE_NEW;
214             }
215         }
216         return violation;
217     }

219     /**
220      * Checks whether a right curly should have a line break before.
221      * @param bracePolicy option for placing the right curly brace.
222      * @param details details for validation.
223      * @return true if a right curly should have a line break before.
224      */
225     private static boolean shouldHaveLineBreakBefore(RightCurlyOption bracePolicy,
226                                                      Details details) {
227         return bracePolicy == RightCurlyOption.SAME
228                 && !hasLineBreakBefore(details.rcurly)
229                 && details.lcurly.getLineNo() != details.rcurly.getLineNo();
230     }
231 
232     /**
233      * Checks that a right curly should be on the same line as the next statement.
234      * @param bracePolicy option for placing the right curly brace
235      * @param details Details for validation
236      * @return true if a right curly should be alone on a line.
237      */
238     private static boolean shouldBeOnSameLine(RightCurlyOption bracePolicy, Details details) {
239         return bracePolicy == RightCurlyOption.SAME
240                 && details.rcurly.getLineNo() != details.nextToken.getLineNo();
241     }
242 
243     /**
244      * Checks that a right curly should be alone on a line.
245      * @param bracePolicy option for placing the right curly brace
246      * @param details Details for validation
247      * @return true if a right curly should be alone on a line.
248      */
249     private static boolean shouldBeAloneOnLine(RightCurlyOption bracePolicy, Details details) {
250         return bracePolicy == RightCurlyOption.ALONE
251                 && !isAloneOnLine(details)
252                 && !isEmptyBody(details.lcurly)
253                 || bracePolicy == RightCurlyOption.ALONE_OR_SINGLELINE
254                 && !isAloneOnLine(details)
255                 && !isSingleLineBlock(details)
256                 && !isAnonInnerClassInit(details.lcurly)
257                 && !isEmptyBody(details.lcurly);
258     }
259 
260     /**
261      * Whether right curly brace starts target source line.
262      * @param details Details of right curly brace for validation
263      * @param targetSourceLine source line to check
264      * @return true if right curly brace starts target source line.
265      */
266     private static boolean isOnStartOfLine(Details details, String targetSourceLine) {
267         return CommonUtils.hasWhitespaceBefore(details.rcurly.getColumnNo(), targetSourceLine)
268                 || details.lcurly.getLineNo() == details.rcurly.getLineNo();
269     }
270 
271     /**
272      * Checks whether right curly is alone on a line.
273      * @param details for validation.
274      * @return true if right curly is alone on a line.
275      */
276     private static boolean isAloneOnLine(Details details) {
277         final DetailAST rcurly = details.rcurly;
278         final DetailAST lcurly = details.lcurly;
279         final DetailAST nextToken = details.nextToken;
280         return rcurly.getLineNo() != lcurly.getLineNo()
281             && rcurly.getLineNo() != nextToken.getLineNo();
282     }
283 
284     /**
285      * Checks whether block has a single-line format.
286      * @param details for validation.
287      * @return true if block has single-line format.
288      */
289     private static boolean isSingleLineBlock(Details details) {
290         final DetailAST rcurly = details.rcurly;
291         final DetailAST lcurly = details.lcurly;
292         final DetailAST nextToken = details.nextToken;
293         return rcurly.getLineNo() == lcurly.getLineNo()
294             && rcurly.getLineNo() != nextToken.getLineNo();
295     }
296 
297     /**
298      * Checks whether lcurly is in anonymous inner class initialization.
299      * @param lcurly left curly token.
300      * @return true if lcurly begins anonymous inner class initialization.
301      */
302     private static boolean isAnonInnerClassInit(DetailAST lcurly) {
303         final Scope surroundingScope = ScopeUtils.getSurroundingScope(lcurly);
304         return surroundingScope.ordinal() == Scope.ANONINNER.ordinal();
305     }

307     /**
308      * Collects validation details.
309      * @param ast detail ast.
310      * @return object that contain all details to make a validation.
311      */
312     // -@cs[JavaNCSS] getDetails() method is a huge SWITCH, it has to be monolithic
313     // -@cs[ExecutableStatementCount] getDetails() method is a huge SWITCH, it has to be monolithic
314     // -@cs[NPathComplexity] getDetails() method is a huge SWITCH, it has to be monolithic
315     private static Details getDetails(DetailAST ast) {
316         // Attempt to locate the tokens to do the check
317         boolean shouldCheckLastRcurly = false;
318         DetailAST rcurly = null;
319         final DetailAST lcurly;
320         DetailAST nextToken;
321 
322         switch (ast.getType()) {
323             case TokenTypes.LITERAL_TRY:
324                 if (ast.getFirstChild().getType() == TokenTypes.RESOURCE_SPECIFICATION) {
325                     lcurly = ast.getFirstChild().getNextSibling();
326                 }
327                 else {
328                     lcurly = ast.getFirstChild();
329                 }
330                 nextToken = lcurly.getNextSibling();
331                 rcurly = lcurly.getLastChild();
332 
333                 if (nextToken == null) {
334                     shouldCheckLastRcurly = true;
335                     nextToken = getNextToken(ast);
336                 }
337                 break;
338             case TokenTypes.LITERAL_CATCH:
339                 nextToken = ast.getNextSibling();
340                 lcurly = ast.getLastChild();
341                 rcurly = lcurly.getLastChild();
342                 if (nextToken == null) {
343                     shouldCheckLastRcurly = true;
344                     nextToken = getNextToken(ast);
345                 }
346                 break;
347             case TokenTypes.LITERAL_IF:
348                 nextToken = ast.findFirstToken(TokenTypes.LITERAL_ELSE);
349                 if (nextToken == null) {
350                     shouldCheckLastRcurly = true;
351                     nextToken = getNextToken(ast);
352                     lcurly = ast.getLastChild();
353                 }
354                 else {
355                     lcurly = nextToken.getPreviousSibling();
356                 }
357                 if (lcurly.getType() == TokenTypes.SLIST) {
358                     rcurly = lcurly.getLastChild();
359                 }
360                 break;
361             case TokenTypes.LITERAL_ELSE:
362             case TokenTypes.LITERAL_FINALLY:
363                 shouldCheckLastRcurly = true;
364                 nextToken = getNextToken(ast);
365                 lcurly = ast.getFirstChild();
366                 if (lcurly.getType() == TokenTypes.SLIST) {
367                     rcurly = lcurly.getLastChild();
368                 }
369                 break;
370             case TokenTypes.CLASS_DEF:
371                 final DetailAST child = ast.getLastChild();
372                 lcurly = child.getFirstChild();
373                 rcurly = child.getLastChild();
374                 nextToken = ast;
375                 break;
376             case TokenTypes.CTOR_DEF:
377             case TokenTypes.STATIC_INIT:
378             case TokenTypes.INSTANCE_INIT:
379                 lcurly = ast.findFirstToken(TokenTypes.SLIST);
380                 rcurly = lcurly.getLastChild();
381                 nextToken = getNextToken(ast);
382                 break;
383             case TokenTypes.LITERAL_DO:
384                 nextToken = ast.findFirstToken(TokenTypes.DO_WHILE);
385                 lcurly = ast.findFirstToken(TokenTypes.SLIST);
386                 if (lcurly != null) {
387                     rcurly = lcurly.getLastChild();
388                 }
389                 break;
390             default:
391                 // ATTENTION! We have default here, but we expect case TokenTypes.METHOD_DEF,
392                 // TokenTypes.LITERAL_FOR, TokenTypes.LITERAL_WHILE only.
393                 // It has been done to improve coverage to 100%. I couldn't replace it with
394                 // if-else-if block because code was ugly and didn't pass pmd check.
395 
396                 lcurly = ast.findFirstToken(TokenTypes.SLIST);
397                 if (lcurly != null) {
398                     // SLIST could be absent if method is abstract,
399                     // and code like "while(true);"
400                     rcurly = lcurly.getLastChild();
401                 }
402                 nextToken = getNextToken(ast);
403                 break;
404         }
405 
406         final Details details = new Details();
407         details.rcurly = rcurly;
408         details.lcurly = lcurly;
409         details.nextToken = nextToken;
410         details.shouldCheckLastRcurly = shouldCheckLastRcurly;
411 
412         return details;
413     }

415     /**
416      * Checks if definition body is empty.
417      * @param lcurly left curly.
418      * @return true if definition body is empty.
419      */
420     private static boolean isEmptyBody(DetailAST lcurly) {
421         boolean result = false;
422         if (lcurly.getParent().getType() == TokenTypes.OBJBLOCK) {
423             if (lcurly.getNextSibling().getType() == TokenTypes.RCURLY) {
424                 result = true;
425             }
426         }
427         else if (lcurly.getFirstChild().getType() == TokenTypes.RCURLY) {
428             result = true;
429         }
430         return result;
431     }

433     /**
434      * Finds next token after the given one.
435      * @param ast the given node.
436      * @return the token which represents next lexical item.
437      */
438     private static DetailAST getNextToken(DetailAST ast) {
439         DetailAST next = null;
440         DetailAST parent = ast;
441         while (next == null) {
442             next = parent.getNextSibling();
443             parent = parent.getParent();
444         }
445         return CheckUtils.getFirstNode(next);
446     }

448     /**
449      * Checks if right curly has line break before.
450      * @param rightCurly right curly token.
451      * @return true, if right curly has line break before.
452      */
453     private static boolean hasLineBreakBefore(DetailAST rightCurly) {
454         final DetailAST previousToken = rightCurly.getPreviousSibling();
455         return previousToken == null
456                 || rightCurly.getLineNo() != previousToken.getLineNo();
457     }
458 
459     /**
460      * Structure that contains all details for validation.
461      */
462     private static class Details {
463         /** Right curly. */
464         private DetailAST rcurly;
465         /** Left curly. */
466         private DetailAST lcurly;
467         /** Next token. */
468         private DetailAST nextToken;
469         /** Should check last right curly. */
470         private boolean shouldCheckLastRcurly;
471     }
