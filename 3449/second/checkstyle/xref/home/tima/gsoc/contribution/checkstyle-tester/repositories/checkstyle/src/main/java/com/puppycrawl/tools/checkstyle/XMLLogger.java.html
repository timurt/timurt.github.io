28  import java.util.ResourceBundle;

30  import com.puppycrawl.tools.checkstyle.api.AuditEvent;
31  import com.puppycrawl.tools.checkstyle.api.AuditListener;

34  import com.puppycrawl.tools.checkstyle.utils.CommonUtils;

36  /**
37   * Simple XML logger.
38   * It outputs everything in UTF-8 (default XML encoding is UTF-8) in case
39   * we want to localize error messages or simply that file names are
40   * localized and takes care about escaping as well.
41  
42   * @author <a href="mailto:stephane.bailliez@wanadoo.fr">Stephane Bailliez</a>
43   */
44  // -@cs[AbbreviationAsWordInName] We can not change it as,
45  // check's name is part of API (used in configurations).
46  public class XMLLogger
47      extends AutomaticBean
48      implements AuditListener {
49      /** Decimal radix. */
50      private static final int BASE_10 = 10;
51  
52      /** Hex radix. */
53      private static final int BASE_16 = 16;
54  
55      /** Some known entities to detect. */
56      private static final String[] ENTITIES = {"gt", "amp", "lt", "apos",
57                                                "quot", };
58  
59      /** Close output stream in auditFinished. */
60      private final boolean closeStream;
61  
62      /** Helper writer that allows easy encoding and printing. */
63      private PrintWriter writer;
64  
65      /**
66       * Creates a new {@code XMLLogger} instance.
67       * Sets the output to a defined stream.
68       * @param outputStream the stream to write logs to.
69       * @param closeStream close oS in auditFinished
70       */
71      public XMLLogger(OutputStream outputStream, boolean closeStream) {
72          setOutputStream(outputStream);
73          this.closeStream = closeStream;
74      }
75  
76      /**
77       * Sets the OutputStream.
78       * @param outputStream the OutputStream to use
79       **/
80      private void setOutputStream(OutputStream outputStream) {
81          final OutputStreamWriter osw = new OutputStreamWriter(outputStream, StandardCharsets.UTF_8);
82          writer = new PrintWriter(osw);
83      }
84  
85      @Override
86      public void auditStarted(AuditEvent event) {
87          writer.println("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
88  
89          final ResourceBundle compilationProperties =
90              ResourceBundle.getBundle("checkstylecompilation", Locale.ROOT);
91          final String version =
92              compilationProperties.getString("checkstyle.compile.version");
93  
94          writer.println("<checkstyle version=\"" + version + "\">");
95      }
96  
97      @Override
98      public void auditFinished(AuditEvent event) {
99          writer.println("</checkstyle>");
100         if (closeStream) {
101             writer.close();
102         }
103         else {
104             writer.flush();
105         }
106     }
107 
108     @Override
109     public void fileStarted(AuditEvent event) {
110         writer.println("<file name=\"" + encode(event.getFileName()) + "\">");
111     }
112 
113     @Override
114     public void fileFinished(AuditEvent event) {
115         writer.println("</file>");
116     }
117 
118     @Override
119     public void addError(AuditEvent event) {
120         if (event.getSeverityLevel() != SeverityLevel.IGNORE) {
121             writer.print("<error" + " line=\"" + event.getLine() + "\"");
122             if (event.getColumn() > 0) {
123                 writer.print(" column=\"" + event.getColumn() + "\"");
124             }
125             writer.print(" severity=\""
126                 + event.getSeverityLevel().getName()
127                 + "\"");
128             writer.print(" message=\""
129                 + encode(event.getMessage())
130                 + "\"");
131             writer.println(" source=\""
132                 + encode(event.getSourceName())
133                 + "\"/>");
134         }
135     }
136 
137     @Override
138     public void addException(AuditEvent event, Throwable throwable) {
139         final StringWriter stringWriter = new StringWriter();
140         final PrintWriter printer = new PrintWriter(stringWriter);
141         printer.println("<exception>");
142         printer.println("<![CDATA[");
143         throwable.printStackTrace(printer);
144         printer.println("]]>");
145         printer.println("</exception>");
146         printer.flush();
147         writer.println(encode(stringWriter.toString()));
148     }
149 
150     /**
151      * Escape &lt;, &gt; &amp; &#39; and &quot; as their entities.
152      * @param value the value to escape.
153      * @return the escaped value if necessary.
154      */
155     public static String encode(String value) {
156         final StringBuilder sb = new StringBuilder();
157         for (int i = 0; i < value.length(); i++) {
158             final char chr = value.charAt(i);
159             switch (chr) {
160                 case '<':
161                     sb.append("&lt;");
162                     break;
163                 case '>':
164                     sb.append("&gt;");
165                     break;
166                 case '\'':
167                     sb.append("&apos;");
168                     break;
169                 case '\"':
170                     sb.append("&quot;");
171                     break;
172                 case '&':
173                     sb.append(encodeAmpersand(value, i));
174                     break;
175                 default:
176                     sb.append(chr);
177                     break;
178             }
179         }
180         return sb.toString();
181     }
182 
183     /**
184      * @param ent the possible entity to look for.
185      * @return whether the given argument a character or entity reference
186      */
187     public static boolean isReference(String ent) {
188         boolean reference = false;
189 
190         if (ent.charAt(0) != '&' || !CommonUtils.endsWithChar(ent, ';')) {
191             reference = false;
192         }
193         else if (ent.charAt(1) == '#') {
194             // prefix is "&#"
195             int prefixLength = 2;
196 
197             int radix = BASE_10;
198             if (ent.charAt(2) == 'x') {
199                 prefixLength++;
200                 radix = BASE_16;
201             }
202             try {
203                 Integer.parseInt(
204                     ent.substring(prefixLength, ent.length() - 1), radix);
205                 reference = true;
206             }
207             catch (final NumberFormatException ignored) {
208                 reference = false;
209             }
210         }
211         else {
212             final String name = ent.substring(1, ent.length() - 1);
213             for (String element : ENTITIES) {
214                 if (name.equals(element)) {
215                     reference = true;
216                     break;
217                 }
218             }
219         }
220         return reference;
221     }
222 
223     /**
224      * Encodes ampersand in value at required position.
225      * @param value string value, which contains ampersand
226      * @param ampPosition position of ampersand in value
227      * @return encoded ampersand which should be used in xml
228      */
229     private static String encodeAmpersand(String value, int ampPosition) {
230         final int nextSemi = value.indexOf(';', ampPosition);
231         final String result;
232         if (nextSemi < 0
233             || !isReference(value.substring(ampPosition, nextSemi + 1))) {
234             result = "&amp;";
235         }
236         else {
237             result = "&";
238         }
239         return result;
240     }
