440     @Test
441     public void testLoadPropertiesIoException() throws Exception {
442         final Class<?>[] param = new Class<?>[1];
443         param[0] = File.class;
444         final Method method = Main.class.getDeclaredMethod("loadProperties", param);
445         method.setAccessible(true);
446         try {
447             if (System.getProperty("os.name").toLowerCase(Locale.ENGLISH).startsWith("windows")) {
448                 // https://support.microsoft.com/en-us/kb/177506 but this only for NTFS
449                 // WindowsServer 2012 use Resilient File System (ReFS), so any name is ok
450                 final File file = new File(File.separator + ":invalid");
451                 if (file.exists()) {
452                     file.delete();
453                 }
454                 method.invoke(null, new File(file.getAbsolutePath()));
455             }
456             else {
457                 method.invoke(null, new File(File.separator + "\0:invalid"));
458             }
459             fail("Exception was expected");
460         }
461         catch (InvocationTargetException ex) {
462             assertTrue(ex.getCause() instanceof CheckstyleException);
463             // We do separate validation for message as in Windows
464             // disk drive letter appear in message,
465             // so we skip that drive letter for compatibility issues
466             assertTrue(ex.getCause().getMessage()
467                     .startsWith("Unable to load properties from file '"));
468             assertTrue(ex.getCause().getMessage().endsWith(":invalid'."));
469         }
470     }
